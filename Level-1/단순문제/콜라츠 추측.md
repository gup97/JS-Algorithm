단순문제

```JS
function solution(num) {
    var answer = 0;
    const collatz =(n)=>{
        if(n===1||answer>=500)return
        answer++;
        if(n%2===0)collatz(n/2)
        else collatz(n*3+1)
    }
    collatz(num)
    return answer<500?answer:-1;
}
```
